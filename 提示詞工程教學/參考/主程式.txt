// --- 主組件 ---

const App = () => {
  const [currentPage, setCurrentPage] = useState('home');
  const [studentName, setStudentName] = useState('');
  const [content, setContent] = useState('');
  const [file, setFile] = useState(null);
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [submitStatus, setSubmitStatus] = useState(null);
  const [searchName, setSearchName] = useState('');
  const [searchResults, setSearchResults] = useState([]);
  const [isSearching, setIsSearching] = useState(false);
  const [adminPassword, setAdminPassword] = useState('');
  const [isAdminAuthenticated, setIsAdminAuthenticated] = useState(false);
  const [loginError, setLoginError] = useState('');

  // --- TTS 功能 ---
  const [playingId, setPlayingId] = useState(null);
  const synth = window.speechSynthesis;

  const handleSpeak = (text, id) => {
    if (playingId === id) {
      synth.cancel();
      setPlayingId(null);
      return;
    }
    synth.cancel();
    const utterance = new SpeechSynthesisUtterance(text);
    utterance.lang = 'zh-TW';
    utterance.onend = () => setPlayingId(null);
    utterance.onstart = () => setPlayingId(id);
    synth.speak(utterance);
  };

  // --- 檔案轉 Base64 ---
  const fileToBase64 = (file) => {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.readAsDataURL(file);
      reader.onload = () => resolve(reader.result.split(',')[1]);
      reader.onerror = (error) => reject(error);
    });
  };

  // --- API 呼叫 ---
  const submitWork = async () => {
    if (!studentName || !content) return;
    setIsSubmitting(true);
    setSubmitStatus('系統正在加密上傳並由 AI 分析評語，請稍候...');

    let fileData = null;
    let fileName = "";
    let mimeType = "";

    if (file) {
      try {
        fileData = await fileToBase64(file);
        fileName = file.name;
        mimeType = file.type;
      } catch (err) {
        console.error("File processing error", err);
      }
    }

    try {
      // GAS POST 必須使用 no-cors 模式，但這會導致回傳被瀏覽器隱藏 (opaque)
      // 我們根據 fetch 是否成功完成來判斷
      await fetch(GAS_API_URL, {
        method: 'POST',
        mode: 'no-cors', 
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          action: 'submit',
          name: studentName,
          content: content,
          fileName: fileName,
          fileData: fileData,
          mimeType: mimeType
        })
      });
      
      setSubmitStatus('作品與檔案已成功送出！請確認信箱與試算表。');
      setStudentName('');
      setContent('');
      setFile(null);
    } catch (error) {
      setSubmitStatus('提交過程發生網路錯誤，請重新嘗試。');
    } finally {
      setIsSubmitting(false);
    }
  };

  const queryReviews = async () => {
    if (!searchName) return;
    setIsSearching(true);
    try {
      const response = await fetch(`${GAS_API_URL}?name=${encodeURIComponent(searchName)}`);
      const data = await response.json();
      setSearchResults(data);
    } catch (error) {
      console.error("Query error:", error);
      setSearchResults([]);
    } finally {
      setIsSearching(false);
    }
  };

  const handleAdminLogin = () => {
    if (adminPassword === '0207') {
      setIsAdminAuthenticated(true);
      setLoginError('');
    } else {
      setLoginError('密碼錯誤，請重新輸入。');
    }
  };

  return (
    <div className="min-h-screen bg-[#FDFDFF] font-sans pb-32 selection:bg-indigo-100 selection:text-indigo-900">
      <nav className="sticky top-0 z-50 bg-white/80 backdrop-blur-lg border-b border-slate-100 h-20 flex items-center">
        <div className="max-w-6xl mx-auto px-6 w-full flex justify-between items-center">
          <div className="flex items-center gap-2 font-black text-2xl text-indigo-600 cursor-pointer hover:opacity-80 transition-opacity" onClick={() => setCurrentPage('home')}>
            <Award size={32} />
            學生作品系統
          </div>
          <div className="hidden md:flex gap-8">
            <button onClick={() => setCurrentPage('home')} className={`text-sm font-black transition-all hover:text-indigo-500 ${currentPage === 'home' ? 'text-indigo-600' : 'text-slate-400'}`}>作品上傳</button>
            <button onClick={() => setCurrentPage('query')} className={`text-sm font-black transition-all hover:text-indigo-500 ${currentPage === 'query' ? 'text-indigo-600' : 'text-slate-400'}`}>評語查詢</button>
            <button onClick={() => setCurrentPage('admin')} className={`text-sm font-black transition-all hover:text-indigo-500 ${currentPage === 'admin' ? 'text-indigo-600' : 'text-slate-400'}`}>管理後台</button>
          </div>
        </div>
      </nav>

      <main className="pt-10 max-w-6xl mx-auto px-4">
        {currentPage === 'home' && (
          <Home 
            studentName={studentName} setStudentName={setStudentName}
            content={content} setContent={setContent}
            file={file} setFile={setFile}
            isSubmitting={isSubmitting} submitStatus={submitStatus}
            submitWork={submitWork}
          />
        )}
        {currentPage === 'query' && (
          <QueryView 
            searchName={searchName} setSearchName={setSearchName}
            isSearching={isSearching} queryReviews={queryReviews}
            searchResults={searchResults} playingId={playingId}
            handleSpeak={handleSpeak}
          />
        )}
        {currentPage === 'admin' && (
          <TeacherPanel 
            isAdminAuthenticated={isAdminAuthenticated} setIsAdminAuthenticated={setIsAdminAuthenticated}
            adminPassword={adminPassword} setAdminPassword={setAdminPassword}
            loginError={loginError} handleAdminLogin={handleAdminLogin}
          />
        )}
      </main>

      {/* 手機導航 */}
      <div className="md:hidden fixed bottom-8 left-8 right-8 h-20 bg-slate-900 rounded-[2.5rem] flex justify-around items-center px-4 z-50 shadow-2xl border border-white/10">
        <button onClick={() => setCurrentPage('home')} className={`text-white transition-opacity ${currentPage === 'home' ? 'opacity-100' : 'opacity-40'}`}><Send size={24} /></button>
        <button onClick={() => setCurrentPage('query')} className={`text-white transition-opacity ${currentPage === 'query' ? 'opacity-100' : 'opacity-40'}`}><Search size={24} /></button>
        <button onClick={() => setCurrentPage('admin')} className={`text-white transition-opacity ${currentPage === 'admin' ? 'opacity-100' : 'opacity-40'}`}><Lock size={24} /></button>
      </div>
    </div>
  );
};

export default App;
